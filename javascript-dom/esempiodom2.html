function report1(tagname) {

	var tagList = document.getElementsByTagName(tagname);
	var s= "";
	for (var i = 0; i < tagList.length; i++)
		 s+= (tagList.item(i).nodeName) + "=" + (tagList.item(i).innerHTML) + "\n";
	return s;
}
