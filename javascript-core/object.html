Creazione di Object. 
var empty = {};						// oggetto vuoto (equivale a new Object)
var point = { x: 10, y: 15};		// oggetto con due proprietà assegnate
var book = {						// oggetto complesso (notare che alcune proprietà sono tra apici)
	"main title": "What If",
	"sub title": "serious scientific answers to absurd hypothetical questions",
	"for": "everyone",
	author: {						// proprietà complessa: è a sua volta un oggetto
		"first-name": "Randall",
		"last-name": "Munroe"
	}
}
